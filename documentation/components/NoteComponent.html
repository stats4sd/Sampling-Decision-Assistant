import { Input } from '@angular/core';

@Component({
  selector: 'note',
  templateUrl: 'note.html'
})
export class NoteComponent {

 @Input('text') text:string;
 @Input() set type(type:string){
   this._type=type
   if(this.icons[type]){
    this.icon = this.icons[type]
   }
   else{
    this.icon = this.type
   }
 }
